787 template<typename IterT, typename IterU = IterT>
788 inline decltype(auto) is_correct(IterT first, IterU last)
789 {
790 using diagnostic_iterator =
791 detail::Diagnostic_iterator<decltype(last)>;
792
793 decltype(last) first_it = first;
794
795 diagnostic_iterator result;
796
797 for (; first_it != last; ++first_it)
798 {
799 if (!is_correct(*first_it))
800 {
801 result = diagnostic_iterator{ first_it, true };
802
803 break;
804 }
805 }
806
807 return result;
808 }

931 template<typename StringT, typename ContainerT, std::enable_if_t<
932 std::is_same_v<std::decay_t<StringT>,
933 std::basic_string<value_type_t<std::decay_t<StringT>>>> &&
934
935 std::is_same_v<value_type_t<std::decay_t<ContainerT>>,
936 std::basic_string<universal_value_type_t<std::decay_t<ContainerT>>>>,
937
938 std::nullptr_t> = nullptr>
939 inline decltype(auto) create_node(StringT&& name, StringT&& value,
940 ContainerT&& attribute_names, ContainerT&& attribute_values)
941 {
942 using symbol_type = std::decay_t<typename StringT::value_type>;
943 using node_type = Node<symbol_type>;
944 using node_pointer = typename Node<symbol_type>::Ptr;
945 using node_attribute = typename Node<symbol_type>::attribute_type;
946 using string_type = typename Node<symbol_type>::string_type;
947
948 node_pointer result = std::make_unique<node_type>(name, value);
949
950 auto attr_names_beg = attribute_names.begin();
951 auto attr_names_end = attribute_names.end();
952
953 auto attr_values_beg = attribute_values.begin();
954 auto attr_values_end = attribute_values.end();
955
956 for (; attr_names_beg != attr_names_end &&
957 attr_values_beg != attr_values_end;
958 ++attr_names_beg, ++attr_values_beg)
959 {
960 result->add_attribute(node_attribute{
961 std::move(*attr_names_beg), std::move(*attr_values_beg) });
962 }
963
964 return result;
965 }

970 std::enable_if_t<
971 std::is_same_v<std::decay_t<StringT>,
972 std::basic_string_view<value_type_t<std::decay_t<StringT>>>>&&
973
974 std::is_same_v<value_type_t<std::decay_t<ContainerT>>,
975 std::basic_string_view<
976 universal_value_type_t<std::decay_t<ContainerT>>>>,
977
978 std::nullptr_t> = nullptr>
979 inline decltype(auto) create_node(StringT&& name, StringT&& value,
980 ContainerT&& attribute_names, ContainerT&& attribute_values)
981 {
982 using symbol_type = std::decay_t<typename StringT::value_type>;
983 using node_type = Node<symbol_type>;
984 using node_pointer = typename Node<symbol_type>::Ptr;
985 using node_attribute = typename Node<symbol_type>::attribute_type;
986 using string_type = typename Node<symbol_type>::string_type;
987
988 node_pointer result = std::make_unique<node_type>(
989 string_type{ name }, string_type{ value });

991 auto attr_names_beg = attribute_names.begin();
992 auto attr_names_end = attribute_names.end();
993
994 auto attr_values_beg = attribute_values.begin();
995 auto attr_values_end = attribute_values.end();
996
997 for (; attr_names_beg != attr_names_end &&
998 attr_values_beg != attr_values_end;
999 ++attr_names_beg, ++attr_values_beg)
1000 {
1001 result->add_attribute(node_attribute{
1002 string_type{ *attr_names_beg },
1003 string_type{ *attr_values_beg} });
1004 }
1005
1006 return result;
1007 }

1011 template<typename ContainerT>
1012 inline decltype(auto) create_root_node_from_container(
1013 ContainerT&& container)
1014 {
1015 using tag_container_type =
1016 typename std::decay_t<ContainerT>::value_type;
1017 using symbol_type = symbol_type_t<tag_container_type>;
1018
1019 using iterator_type =
1020 iterator_t<typename std::decay_t<ContainerT>::value_type::
1021 child_container_type>;
1022
1023 using node_type = Node<symbol_type>;
1024 using node_pointer = typename Node<symbol_type>::Ptr;
1025 using tag_type = Tag_range<iterator_type, node_pointer>;
1026
1027 using std::swap;
1028
1029 node_pointer result{ nullptr };
1030
1031 if (container.size())
1032 {
1033 auto&& source_root_tag = container.top();
1034
1035 std::stack<tag_type> tag_nodes;
1036
1037 node_pointer tag_node = create_node(
1038 std::move(source_root_tag.name),
1039 std::move(source_root_tag.value),
1040 source_root_tag.attribute_names,
1041 source_root_tag.attribute_values);
1042
1043 tag_nodes.push(tag_type{
1044 source_root_tag.childs.begin(),
1045 source_root_tag.childs.end(),
1046 std::move(tag_node) });
1047
1048 auto first = tag_nodes.top().first;
1049 auto last = tag_nodes.top().last;
1050
1051 while (first != last)
1052 {
1053 node_pointer temp_tag = create_node(
1054 std::move((*first).name),
1055 std::move((*first).value),
1056 (*first).attribute_names,
1057 (*first).attribute_values);
1058
1059 if ((*first).childs.size())
1060 {
1061 tag_nodes.top().first = first;
1062
1063 tag_nodes.push(tag_type{ (*first).childs.begin(),
1064 (*first).childs.end(), std::move(temp_tag) });
1065
1066 first = tag_nodes.top().first;
1067 last = tag_nodes.top().last;
1068 }
1069 else
1070 {
1071 tag_nodes.top().node->add_child(std::move(temp_tag));
1072
1073 ++first;
1074 }
1075
1076 while (first == last && tag_nodes.size() > 1)
1077 {
1078 auto temp = std::move(tag_nodes.top().node);
1079 tag_nodes.pop();
1080 tag_nodes.top().node->add_child(std::move(temp));
1081
1082 ++tag_nodes.top().first;
1083
1084 first = tag_nodes.top().first;
1085 last = tag_nodes.top().last;
1086 }
1087 }
1088
1089 swap(result, tag_nodes.top().node);
1090 }
1091
1092 return result;
