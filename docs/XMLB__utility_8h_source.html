53 template<typename CharT>
54 inline float to_float(std::basic_string_view<CharT> str)
55 {
56 using symbol_type = CharT;
57 using size_type =
58 typename std::basic_string_view<symbol_type>::size_type;
59
60 float result = 0.f;
61
62 if (str.size())
63 {
64 const unsigned int base = 10;
65 const unsigned int start_ascii_for_dec = 48;
66 const unsigned int dot_count = 1;
67 const symbol_type dot_symbol = '.';
68 const symbol_type minus_symbol = '-';
69
70 auto found_minus = str.find_first_of(minus_symbol);
71 auto found_dot = str.find_first_of(dot_symbol);
72
73 size_type head_count = 0;
74 size_type tail_count = 0;
75
76 if (found_dot != std::basic_string_view<CharT>::npos)
77 {
78 head_count = found_dot;
79 tail_count = str.size() - dot_count - head_count;
80 }
81 else
82 {
83 head_count = str.size();
84 }
85
86 if (found_minus != std::basic_string_view<CharT>::npos)
87 {
88 --head_count;
89 }
90
91 float head = 0.f;
92 float tail = 0.f;
93
94 float start_base = 1.f;
95 size_type digit_position = found_dot - 1;
96
97 while (head_count)
98 {
99 float digit = static_cast<float>(str[digit_position] -
100 start_ascii_for_dec);
101
102 head += digit * start_base;
103 start_base *= base;
104
105 --head_count;
106 --digit_position;
107 }
108
109 start_base = base;
110 digit_position = found_dot + 1;
111
112 while (tail_count)
113 {
114 float digit = static_cast<float>(str[digit_position] -
115 start_ascii_for_dec);
116
117 tail += digit > 0 ? digit / start_base : 0.f;
118 start_base *= base;
119
120 --tail_count;
121 ++digit_position;
122 }
123
124 result = head + tail;
125
126 if (found_minus != std::basic_string_view<CharT>::npos)
127 {
128 result *= -1.f;
129 }
130 }
131
132 return result;
133 }

153 template<typename CharT>
154 inline std::basic_string<CharT> to_string(float value, int base = 10)
155 {
156 using std::begin;
157 using std::end;
158
159 using symbol_type = CharT;
160 using string_type = std::basic_string<symbol_type>;
161 using size_type = typename std::basic_string<symbol_type>::size_type;
162
163 const unsigned int digit_count = 7;
164 const unsigned int dot_count = 1;
165 const unsigned int sign_count = 1;
166 const unsigned int start_ascii_for_hex = 87;
167 const unsigned int start_ascii_for_dec = 48;
168 const symbol_type dot_symbol = '.';
169 const symbol_type minus_symbol = '-';
170
171 string_type result;
172 int result_char_count;
173 size_type head_count = 1;
174 int digit_position = 1;
175
176 // Корректируем размер результирующей строки и значение вспомагательной
177 // позиции цыфры
178
179 if (value < 0)
180 {
181 result_char_count = digit_count + dot_count + sign_count;
182 result.resize(result_char_count,
183 static_cast<char>(start_ascii_for_dec));
184
185 digit_position = 0;
186 }
187 else
188 {
189 result_char_count = digit_count + dot_count;
190 result.resize(result_char_count,
191 static_cast<char>(start_ascii_for_dec));
192 }
193
194 // Инициализируем целую и дробные части входного числа
195
196 int head = static_cast<int>(std::abs(value));
197
198 {
199 int temp_base = base;
200
201 while (head > temp_base)
202 {
203 temp_base *= base;
204 ++head_count;
205 }
206 }
207
208 float tail_raw = (std::abs(value) - static_cast<float>(head));
209 float tail_count = static_cast<float>(
210 std::pow(base, digit_count - head_count));
211
212 int tail = static_cast<int>(std::round(tail_raw * tail_count));
213
214 // Конвертация целой части в строку
215
216 do
217 {
218 int digit = head % base;
219 char symbol = static_cast<char>(digit + start_ascii_for_dec);
220
221 result[head_count - digit_position] = symbol;
222 --result_char_count;
223 ++digit_position;
224
225 head /= base;
226
227 } while (head && result_char_count >= 0);
228
229 // Добавляем знак, если он нужен, и точку
230
231 if (value < 0)
232 {
233 result.front() = minus_symbol;
234 --result_char_count;
235 }
236
237 result[result.size() - result_char_count] = dot_symbol;
238 --result_char_count;
239
240 // Конвертация дробной части в строку
241
242 digit_position = 1;
243
244 while (tail && result_char_count > 0)
245 {
246 int digit = tail % base;
247 char symbol = static_cast<char>(digit + start_ascii_for_dec);
248
249 result[result.size() - digit_position] = symbol;
250 --result_char_count;
251 ++digit_position;
252
253 tail /= base;
254
255 };
256
257 return result;
258 }

278 template<typename CharT>
279 inline std::basic_string<CharT> cut_doc_version(float version)
280 {
281 using symbol_type = CharT;
282 using size_type = typename std::basic_string<symbol_type>::size_type;
283
284 auto result = to_string<symbol_type>(version);
285
286 const symbol_type dot_symbol = '.';
287 const symbol_type zero_digit_symbol = '0';
288
289 auto found_dot = result.find_first_of(dot_symbol);
290
291 size_type not_zero_digit_index = std::basic_string<symbol_type>::npos;
292
293 for (auto index = found_dot + 1; index != result.size(); ++index)
294 {
295 if (result[index] != zero_digit_symbol)
296 {
297 not_zero_digit_index = index;
298 }
299 }
300
301 if (not_zero_digit_index != std::basic_string<symbol_type>::npos)
302 {
303 result.erase(not_zero_digit_index + 1);
304 }
305 else
306 {
307 result.erase(found_dot + 2);
308 }
309
310 result.shrink_to_fit();
311
312 return result;
313 }

337 template<typename CharT>
338 inline std::size_t symbols_count(
339 const XMLB::Document<CharT>& document, bool fill_status = true,
340 bool line_break_status = true)
341 {
342 using symbol_type = CharT;
343 std::size_t result = 0;
344
345 constexpr auto kSpace = 1;
346 constexpr auto kOpen_tag = 1;
347 constexpr auto kClose_tag = 1;
348 constexpr auto kOpen_attr = 1;
349 constexpr auto kClose_attr = 1;
350 constexpr auto kEqual_attr = 1;
351 constexpr auto kDoc_info = 1;
352 constexpr auto kDoc_last_info = 1;
353 constexpr auto kLine_break = line_break_status ? 1 : 0;
354
355 constexpr auto kXml = 3;
356 constexpr auto kVersion = 7;
357 constexpr auto kEncoding = 8;
358
359 result += kOpen_tag;
360 result += kDoc_info;
361 result += kXml;
362
363 result += kSpace;
364 result += kVersion;
365 result += kEqual_attr;
366 result += kOpen_attr;
367 result += cut_doc_version<symbol_type>(document.get_version()).size();
368 result += kClose_attr;
369
370 result += kSpace;
371 result += kEncoding;
372 result += kEqual_attr;
373 result += kOpen_attr;
374 result += document.get_encoding_type().size();
375 result += kClose_attr;
376
377 result += kDoc_last_info;
378 result += kClose_tag;
379 result += kLine_break;
380
381 result += symbols_count(document.cbegin(), document.cend(),
382 fill_status, line_break_status);
383
384 return result;
385 }

410 template<typename CharT>
411 inline std::size_t symbols_count(
412 detail::Node_const_iterator<CharT> first,
413 detail::Node_const_iterator<CharT> last,
414 bool fill_status = true,
415 bool line_break_status = true)
416 {
417 using const_iterator = detail::Node_const_iterator<CharT>;
418
419 std::size_t result = 0;
420
421 if (first == last)
422 {
423 return result;
424 }
425
426 constexpr auto kFill = fill_status ? 1 : 0;
427 constexpr auto kSpace = 1;
428 constexpr auto kOpen_tag = 1;
429 constexpr auto kClose_tag = 1;
430 constexpr auto kSingle_tag = 1;
431 constexpr auto kLast_tag = 1;
432 constexpr auto kOpen_attr = 1;
433 constexpr auto kClose_attr = 1;
434 constexpr auto kEqual_attr = 1;
435 constexpr auto kLine_break = line_break_status ? 1 : 0;
436 constexpr auto kTab = 1;
437 constexpr auto kCarriage = 1;
438
439 //Контейнер итераторова, чтобы правильно закрывать теги с потомками
440 std::stack<const_iterator> node_groups;
441
442 for (; first != last; ++first)
443 {
444 //Проверяем, пустой ли стэк с итераторома-группами
445 if (!node_groups.empty())
446 {
447 //Если текущее значение табуляции равно с последним
448 //добавленным в стэк итератором, то заносим в файл
449 //закрытие предыдущего тега
450 while (first.get_offset() <= node_groups.top().get_offset())
451 {
452 //Записываем нужное количество табов для тега
453 result += node_groups.top().get_offset() * kFill;
454
455 result += kOpen_tag;
456 result += kLast_tag;
457 result += node_groups.top()->get_name().size();
458 result += kClose_tag;
459 result += kLine_break;
460
461 node_groups.pop();
462 }
463 }
464
465 //Если у текущего тега есть под-теги, то добавляем его
466 //в стек в качестве итератора-группы
467 if (first->child_size())
468 {
469 node_groups.push(first);
470 }
471
472 //Устанавливаем нужное количество табов для тега
473 result += first.get_offset() * kFill;
474
475 //Заносим в файл сам тег и его информацию
476 result += kOpen_tag;
477 result += first->get_name().size();
478
479 //Если у XML тега есть аттрибуты, то записываем в файл
480 //их имена и значения
481 if (first->attr_size())
482 {
483 for (auto at_it = first->attr_begin(),
484 at_end = first->attr_end();
485 at_it != at_end;
486 ++at_it)
487 {
488 result += kSpace;
489
490 result += at_it->name.size();
491 result += kEqual_attr;
492
493 result += kOpen_attr;
494 result += at_it->value.size();
495 result += kClose_attr;
496 }
497 }
498
499 //Если у XML тега нет дочерних узлов и есть значение, то
500 //записываем это значение и доп. символы по шаблону
501 if (!first->child_size() && first->get_value().size())
502 {
503 result += kClose_tag;
504 result += first->get_value().size();
505
506 result += kOpen_tag;
507 result += kLast_tag;
508
509 result += first->get_name().size();
510 result += kClose_tag;
511 }
512 //Если у XML тега нет дочерних узлов и нет значения, то
513 //записываем доп. символы по шаблону одиночного XML тега
514 else if (!first->child_size())
515 {
516 result += kSpace;
517 result += kSingle_tag;
518 result += kClose_tag;
519 }
520 //В противном случае, просто записываем в файл символ
521 //закрытия XML тега
522 else
523 {
524 result += kClose_tag;
525 }
526
527 result += kLine_break;
528 }
529
530 //Если остались ещё незакрытые теги, то заносим в файл их закрытие
531 while (!node_groups.empty())
532 {
533 //Записываем нужное количество табов для тега
534 result += node_groups.top().get_offset() * kFill;
535
536 result += kOpen_tag;
537 result += kLast_tag;
538 result += node_groups.top()->get_name().size();
539 result += kClose_tag;
540 result += kLine_break;
541
542 node_groups.pop();
543 }
544
545 return result;
546 }

570 template<typename CharT>
571 inline std::size_t symbols_count(const Node<CharT>& node,
572 bool fill_status = true,
573 bool line_break_status = true)
574 {
575 using const_iterator = detail::Node_const_iterator<CharT>;
576
577 std::size_t result = 0;
578
579 constexpr auto kFill = fill_status ? 1 : 0;
580 constexpr auto kSpace = 1;
581 constexpr auto kOpen_tag = 1;
582 constexpr auto kClose_tag = 1;
583 constexpr auto kSingle_tag = 1;
584 constexpr auto kLast_tag = 1;
585 constexpr auto kOpen_attr = 1;
586 constexpr auto kClose_attr = 1;
587 constexpr auto kEqual_attr = 1;
588 constexpr auto kLine_break = line_break_status ? 1 : 0;
589 constexpr auto kTab = 1;
590 constexpr auto kCarriage = 1;
591
592 result += kOpen_tag;
593 result += node.get_name().size();
594
595 //Если у XML тега есть аттрибуты, то записываем в файл
596 //их имена и значения
597 if (node.attr_size())
598 {
599 for (auto at_it = node.attr_begin(),
600 at_end = node.attr_end();
601 at_it != at_end;
602 ++at_it)
603 {
604 result += kSpace;
605
606 result += at_it->name.size();
607 result += kEqual_attr;
608
609 result += kOpen_attr;
610 result += at_it->value.size();
611 result += kClose_attr;
612 }
613 }
614
615 //Если у XML тега нет дочерних узлов и есть значение, то
616 //записываем это значение и доп. символы по шаблону
617 if (!node.child_size() && node.get_value().size())
618 {
619 result += kClose_tag;
620 result += node.get_value().size();
621
622 result += kOpen_tag;
623 result += kLast_tag;
624
625 result += node.get_name().size();
626 result += kClose_tag;
627 }
628 //Если у XML тега нет дочерних узлов и нет значения, то
629 //записываем доп. символы по шаблону одиночного XML тега

642 result += kLine_break;
643 result += kFill;
644
645 result += kOpen_tag;
646 result += kLast_tag;
647 result += node.get_name().size();
648 result += kClose_tag;
649 }
650
651 result += symbols_count(node.begin(), node.end(),
652 fill_status, line_break_status);
653
654 return result;
