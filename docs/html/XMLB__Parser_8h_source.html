180 template<typename InputT, typename DecorT, typename DataT>
181 template<typename CharT>
182 inline bool Parser<InputT, DecorT, DataT>::parse(CharT&& symbol,
183 const DecorT& decorator)
184 {
185 bool result = true;
186
187 m_current_state = m_functors[m_current_state](
188 symbol, m_data, decorator);
189
190 if (m_current_state >= m_functors.size() || !m_functors.size())
191 {
192 result = false;
193 }
194
195 return result;
196 }

218 template<typename InputT, typename DecorT, typename DataT>
219 template<typename FunctorT>
220 inline void Parser<InputT, DecorT, DataT>::add_functor(state_type state,
221 FunctorT functor)
222 {
223 if (state < m_functors.size())
224 {
225 m_functors[state] = std::move(functor);
226 }
227 else
228 {
229 m_functors.emplace_back(functor);
230 }
231 }
