646 template<typename CharT>
647 inline Node<CharT>::Node(Node<CharT>&& node) noexcept
648 :m_name{ std::move(node.m_name) },
649 m_value{ std::move(node.m_value) },
650 m_attributes{ std::move(node.m_attributes) },
651 m_childs{ std::move(node.m_childs) },
652 m_size{ 0 }
653 {
654 m_tree_node.element = this;
655
656 node.m_tree_node.next = nullptr;
657 node.m_tree_node.prev = nullptr;
658 node.m_tree_node.parent = nullptr;
659
660 //Конектим первый дочерний элемент с текущим и изменяем им родителя на
661 //текущий элемент
662 if (m_childs.size())
663 {
664 m_tree_node.next = &m_childs.front()->m_tree_node;
665 m_childs.front()->m_tree_node.prev = &m_tree_node;
666
667 for (auto&& elem : m_childs)
668 {
669 elem->m_tree_node.parent = &m_tree_node;
670 }
671 }
672
673 auto last_child = find_last_tree_node();
674
675 if (last_child != &m_tree_node)
676 {
677 last_child->next = &m_tree_node;
678 m_tree_node.prev = last_child;
679 }
680 }

1117 template<typename CharT>
1118 inline void Node<CharT>::swap(node_type& node) noexcept
1119 {
1120 using std::swap;
1121
1122 //Получаем необходимые начальные узлы текущего элемента
1123 //start -> swap_element -> end
1124
1125 auto lhs_start = m_tree_node.prev;
1126 auto lhs_end = find_last_tree_node()->next;
